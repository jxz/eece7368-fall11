   51   // SINGLE INITIATOR THREAD
   52   //
   53   // This thread is a simple writer and reader thread.  It is used when the 'TLM
   54   // Testbench' and 'TLM Generation' options indicate a simple target component.
   55   // It will write a single input data set, wait for the algorithm processing to
   56   // be complete, then read the output data set for comparison with the Simulink
   57   // data.  It does not use buffering or CSR buffer status.
   58   //
   59   // If you choose more complex 'TLM Generation' options, such as buffering and
   60   // inclusion of a command/status register, a separate writer and reader thread
   61   // will be created for use by the testbench.
   62   // ------------------------------------------------------------------------
   63   void model_jxz_tlm_tb::singleInitiatorThread(void)
   64   {
   65     m_utils.printMsg(TLMG_PRINT_START_OF_SIM,
   66                      "## Start of vectors from MAT file.  Will display '.' for every 100 vectors played.\n");
   67   
   68     // ------------------------------------------------------
   69     // Set core TLM component timing mode: Untimed or Timed.
   70     // ------------------------------------------------------
   71     // Set whether the target should return delays based on the specified timing
   72     // parameters in the 'TLM Generation' options or to just return '0' for all
   73     // transaction and algorithm delays.
   74     //
   75     // In your TLM environment, this can be dynamically programmed during
   76     // simulation execution, but for this testbench we set it just once at the
   77     // beginning of the simulation based on the 'TLM Testbench' configset
   78     // options.
   79     //
   80     // m_backdoorcfg->SetTimingParam(mw_backdoorcfg_IF::UNTIMED);
   81     m_backdoorcfg->SetTimingParam(mw_backdoorcfg_IF::TIMED);
   82   
   83     // The timing mode also needs to be set up in a testbench support object.
   84     //
   85     // With timing  Enable quantum  Testbench timing mode
   86     // -----------  --------------  ---------------------
   87     //     No            X          Untimed
   88     //     Yes           No         FunctionallyTimed
   89     //     Yes           Yes        LooselyTimed
   90     //
   91     // Untimed           : Sync to '0' delays after every transaction and for
   92     //                     every timed wait.
   93     // Functionally Timed: Sync to the delays returned from the target after
   94     //                     every transaction and use non-zero times for timed
   95     //                     waits.
   96     // Loosely Timed     : Track returned delays from the target by using a
   97     //                     tlm_quantumkeeper and sync when the quantum has been
   98     //                     reached.  Occassionally, for proper operation of the
   99     //                     initiator threads, perform synchronization-on-demand
  100     //                     prior to reaching the quantum.
  101     //
  102     // mw_sync_tb       sync(m_utils, mw_sync_tb::Untimed);
  103     mw_sync_tb sync(m_utils, mw_sync_tb::FunctionallyTimed);
  104   
  105     // mw_sync_tb       sync(m_utils, mw_sync_tb::LooselyTimed, m_quantumTime);
  106   
  107     // ------------------------------------------------------
  108     // Instantiate thread-specific testbench support objects.
  109     // ------------------------------------------------------
  110     // Each initiator thread needs to keep track of their local time
  111     // independently.  Here, we instantiate the main time-dependent testbench
  112     // helper objects for TLM transactions and a command/status register
  113     // interface.
  114     //
  115     // mw_tr_tb  : Interface for executing TLM transactions based on current tb
  116     //             timing mode.
  117     // mw_csr_tb : Interface for executing command/status register TLM
  118     //             transactions and higher-level CSR functions such as polling.
  119     //
  120     mw_tr_tb tr(m_utils, sync, m_tlm2_isocket);
  121     mw_csr_tb csr(m_utils, sync, tr, COMSTAT_REG_ADDR, m_mutex);
  122     bool endOfVectors = false;
  123     while (!endOfVectors) {
  124       endOfVectors = m_vecmat.getNextInput();
  125       if (!endOfVectors) {
  126       } else {
  127         break;
  128       }
  129   
  130       // --------------------------------------------------
  131       // Wait until there is data to read.
  132       // --------------------------------------------------
  133       // We can wait for data ready by either polling the INTR bit in the CSR,
  134       // or by waiting for the IRQ signal attached to the m_int_ip port of the
  135       // testbench.  If neither is available, we can only do a timed wait.
  136       //
  137       // Using IRQ:
  138       sync.syncToDataReady(m_int_ip);
  139   
  140       //
  141       // Using CSR:
  142       // m_utils.printMsg(TLMG_PRINT_SYNC_TO_FUNCS,
  143       //         "##\t polling status bit in register for data ready in output buffer...\n");
  144       // bool sawTimeout = csr.pollForStatus(INT_STAT_BIT_MASK, INT_STAT_BIT_MASK, csreg,
  145       //                                     1000, sc_time(1000, SC_NS));
  146       // if (sawTimeout) {
  147       //     m_utils.printErr(TLMG_ERR_SYNC_TIMEOUT,
  148       //         "## ERROR: Timed-out waiting for buffer non-empty status.\n");
  149       //     break;
  150       // }
  151   
  152       // We compare the actual results with the expected results from the
  153       // Simulink vector generation.
  154       //
  155       m_vecmat.putNextActualOutput();    // write actual output to MAT file
  156       endOfVectors = m_vecmat.getNextExpectedOutput();// read expeced output from MAT file
  157       m_vecmat.compareResult();
  158     }
  159   
  160     // ------------------------------------------------------
  161     // Simulation summary.
  162     // ------------------------------------------------------
  163     // We cannot check buffer status bits without a command/status register.
  164     sync.syncToLocalTime();
  165     if (!endOfVectors) {
  166       m_utils.printMsg(TLMG_PRINT_END_OF_SIM,
  167                        "## Error condition caused pre-mature end to vector replay.  Terminating simulation.\n");
  168       m_vecmat.terminate();
  169       sc_stop();
  170     } else {
  171       m_utils.printMsg(TLMG_PRINT_END_OF_SIM,
  172                        "## end of data...Terminating initiator thread.\n");
  173     }
  174   
  175     m_vecmat.terminate();
  176     sc_stop();
  177   }

  180   // initialize all of the port info members:
  181   // m_inFieldInfo, m_actOutFieldInfo, m_expOutFieldInfo
  182   //
  183   void model_jxz_tlm_tb::initialize()
  184   {
  185     // --------------------------------
  186     // initialize the MAT file fields:
  187     //   pmxa   : pointer to main mxArray in MAT File
  188     //   pslf   : pointer to specific field of that mxArray
  189     //   sldsize: the size of the data mxArray pointer for that field
  190     // --------------------------------
  191     m_vecmat.initialize(m_inFieldInfo, m_actOutFieldInfo, m_expOutFieldInfo);
  192   
  193     // --------------------------------
  194     // initialize the C datatype fields:
  195     //   pcf    : pointer to the field in the C structure
  196     //   cfsize : size of the field in the C structure
  197     //   tlmaddr: TLM address of field in the C structure
  198     // --------------------------------
  199   
  200     // initialize the structure for info about the input data set
  201     //
  202   
  203     // initialize the structure for the info about the output data set
  204     //
  205   
  206     // same with the expected output data set--we use this when doing a data
  207     // comparison.
  208     //
  209   
  210     // Since we have generated a TLM with a single input and output address, we do
  211     // not need to initialize a structure to hold the TLM addresses for
  212     // individual inputs and outputs.  We will directly specify the appropriate
  213     // address argument when sending TLM transactions.
  214     //
  215     // Initialize the input TLM addresses:  not needed
  216     // Initialize the output TLM addresses: not needed
  217     //
  218   
  219     // ------------------------------------------------------
  220     // Adjust global quantum if necessary
  221     // ------------------------------------------------------
  222     // When running a temporally-decoupled simulation, it is important that the
  223     // global quantum be chosen carefully to avoid synchronization issues
  224     // between the various threads in the system.  It is possible to choose a
  225     // value that is too coarse or too fine for the system in which the core
  226     // component is being integrated.  Since we are using a generic
  227     // synchronization scheme between the reader and writer threads, we adjust the
  228     // quantum so that proper synchronization should occur.
  229     //
  230     // In your TLM environment, you should choose a value that makes sense for
  231     // your system.
  232     //
  233     m_quantumTime = QUANTUM_TIME;
  234     if (QUANTUM_TIME > OUT_BUFFER_DEPTH*ALG_PROC_TIME) {
  235       m_quantumTime = OUT_BUFFER_DEPTH*ALG_PROC_TIME;
  236     }
  237   
  238     // ------------------------------------------------------
  239     // Adjust buf write count if necessary
  240     // ------------------------------------------------------
  241     // When running a multi-threaded simulation, it is necessary to ensure there
  242     // are no buffer overflow or underflow scenarios.  Since we are using a generic
  243     // synchronization scheme between the reader and writer threads, we adjust
  244     // the buffer write count so that the writerThread will not cause an overflow.
  245     //
  246     // In your TLM environment, you should construct initiators and their
  247     // write and read data rates such that overflow and underflow do not happen.
  248     //
  249     m_bufWriteCount = (IN_BUFFER_DEPTH <= OUT_BUFFER_DEPTH) ?
  250       IN_BUFFER_DEPTH : OUT_BUFFER_DEPTH;
  251   }
