   30   void mw_utils_tb::printMsg(int level, const char * fmt, ...)
   31   {
   32     if (level & m_currVerbosity) {
   33       // prepend thread name and time to message
   34   
   35       // thread name:
   36       sc_process_handle ph = sc_get_current_process_handle();
   37       string thrName(ph.name());
   38       size_t shortNameIdx = thrName.find_last_of('.');
   39       string shortName = thrName.substr(shortNameIdx+1);
   40       size_t isReader = shortName.find("readerThread");
   41   
   42       // sim time:
   43       const sc_time & simtime = sc_time_stamp();
   44       static char buf[512];              // FIXME: detect err on overrun
   45       va_list args;
   46       va_start(args, fmt);
   47       vsprintf(buf, fmt, args);
   48       va_end(args);
   49       if (isReader != string::npos) {
   50         printf("[%10s] (%10s) \t\t%s", (simtime.to_string()).c_str(),
   51                shortName.c_str(), buf);
   52       } else {
   53         printf("[%10s] (%10s) %s", (simtime.to_string()).c_str(), shortName.c_str(),
   54                buf);
   55       }
   56     }
   57   }

  130   //      tlmg_inXXX
  131   //          Data (buffer of elem data type)
  132   //          SampleInfo
  133   //              elemDataType
  134   //              (etc)

  136   //              numerictype

  141   void mw_matfile_tb::initializeInMatFieldInfo(
  142     mw_portinfovec_tb & inFieldInfo,
  143     mw_portinfovec_tb & expOutFieldInfo)
  144   {
  145     if (openMat()) {
  146       m_utils.markError(TLMG_ERR_MAT_INIT);
  147       return;
  148     }
  149   
  150     if (readSclog()) {
  151       m_utils.markError(TLMG_ERR_MAT_INIT);
  152       return;
  153     }
  154   
  155     int numfields = mxGetNumberOfFields(m_sclog);
  156     int in_dcount = 0;
  157     int out_dcount = 0;
  158     SignalDirectionEnumT sigdir;
  159     for (int i=0; i<numfields; i++) {
  160       // ----------------------------------------
  161       // get fieldname and direction
  162       // ----------------------------------------
  163       const char * fname = mxGetFieldNameByNumber(m_sclog, i);
  164       if (!strncmp("tlmg_in", fname, 7)) {
  165         sigdir = INPUT_SIG;
  166         m_utils.printMsg(TLMG_PRINT_VEC_INIT,
  167                          "## found input field %s at %s fieldnum %d\n", fname,
  168                          m_matVar, i);
  169       } else if (!strncmp("tlmg_out", fname, 8)) {
  170         sigdir = OUTPUT_SIG;
  171         m_utils.printMsg(TLMG_PRINT_VEC_INIT,
  172                          "## found output field %s at %s fieldnum %d\n", fname,
  173                          m_matVar, i);
  174       } else {
  175         m_utils.printMsg(TLMG_PRINT_VEC_INIT,
  176                          "## skipping unknown field %s in var %s\n", fname,
  177                          m_matVar);
  178         continue;
  179       }
  180   
  181       // -------------------------------------------------------
  182       // get hold of Data field and its underlying data buffer
  183       // -------------------------------------------------------
  184       mxArray * pmxa_field;
  185       if ((pmxa_field = mxGetFieldByNumber(m_sclog, 0, i)) == NULL) {
  186         m_utils.printErr(TLMG_ERR_MAT_INIT,
  187                          "## ERROR: %s.tlmg_XX var could not be read.\n",
  188                          m_matVar);
  189         return;                          //return(1);
  190       }
  191   
  192       int fnum;
  193       if ((fnum = mxGetFieldNumber(pmxa_field, "Data")) == -1) {
  194         m_utils.printErr(TLMG_ERR_MAT_INIT,
  195                          "## ERROR: %s.tlmg_XX var did not have Data field.\n",
  196                          m_matVar);
  197         return;                          //return(1);
  198       }
  199   
  200       mxArray * pmxa_dfield;
  201       if ((pmxa_dfield = mxGetFieldByNumber(pmxa_field, 0, fnum)) == NULL) {
  202         m_utils.printErr(TLMG_ERR_MAT_INIT,
  203                          "## ERROR: %s.tlmg_XX.Data var could not be read.\n",
  204                          m_matVar);
  205         return;                          //return(1);
  206       }
  207   
  208       if (sigdir == INPUT_SIG) {
  209         inFieldInfo[in_dcount].pmxa = pmxa_dfield;
  210         inFieldInfo[in_dcount].pslf = (char *)mxGetData(pmxa_dfield);
  211         inFieldInfo[in_dcount].sldsize =
  212           mxGetNumberOfElements(pmxa_dfield) * mxGetElementSize(pmxa_dfield);
  213         in_dcount++;
  214       } else {
  215         expOutFieldInfo[out_dcount].pmxa = pmxa_dfield;
  216         expOutFieldInfo[out_dcount].pslf = (char *)mxGetData(pmxa_dfield);
  217         expOutFieldInfo[out_dcount].sldsize =
  218           mxGetNumberOfElements(pmxa_dfield) * mxGetElementSize(pmxa_dfield);
  219         out_dcount++;
  220       }
  221     }
  222   
  223     if (in_dcount != m_numInFields) {
  224       m_utils.printErr(TLMG_ERR_MAT_INIT,
  225                        "## found %d input fields in var %s, expected %d\n",
  226                        in_dcount, m_matVar, NUM_IN_PORTS);
  227     }
  228   
  229     if (out_dcount != m_numOutFields) {
  230       m_utils.printErr(TLMG_ERR_MAT_INIT,
  231                        "## found %d output fields in var %s, expected %d\n",
  232                        out_dcount, m_matVar, NUM_OUT_PORTS);
  233     }
  234   
  235     m_utils.printMsg(TLMG_PRINT_VEC_INIT,
  236                      "## setup %d input data fields, %d output data fields\n",
  237                      in_dcount,
  238                      out_dcount);
  239   }

  241   // ------------------------------------------------------------------------
  242   // expected structure of the MAT file info. . .
  243   //      tlmg_outXXX
  244   //          Data (buffer of elem data type)
  245   //
  246   void mw_matfile_tb::initializeOutMatFieldInfo(
  247     mw_portinfovec_tb & expOutFieldInfo,
  248     mw_portinfovec_tb & actOutFieldInfo)
  249   {
  250     if (openMat()) {
  251       m_utils.markError(TLMG_ERR_MAT_INIT);
  252       return;
  253     }
  254   
  255     const char * topFields[] = {
  256       "tlmg_outX"
  257     };
  258   
  259     mwSize dims[2] = { 1, 1 };
  260   
  261     m_sclog = mxCreateStructArray(2, dims, m_numOutFields, topFields);
  262     if (m_sclog == NULL) {
  263       m_utils.printErr(TLMG_ERR_MAT_INIT,
  264                        "## ERROR: Could not create top-level mx struct for MAT file %s\n",
  265                        m_filename);
  266       exit(-1);                          //return(1);
  267     }
  268   
  269     const char * outStructFieldNames[] = { "Data" };// may add DataTypeInfo later.
  270   
  271     mxArray * paOutStruct;
  272     for (int i=0; i<m_numOutFields; i++) {
  273       // create Data array -- should match expected data out. . .
  274       dims[1] = expOutFieldInfo[i].sldsize;
  275       actOutFieldInfo[i].sldsize = dims[1];
  276       actOutFieldInfo[i].pmxa = mxCreateNumericArray(2, dims, mxUINT8_CLASS,
  277         mxREAL);
  278       if (actOutFieldInfo[i].pmxa == NULL) {
  279         m_utils.printErr(TLMG_ERR_MAT_INIT,
  280                          "## ERROR: Could not create mx numeric array for MAT file %s\n",
  281                          m_filename);
  282         exit(-1);                        //return(1);
  283       }
  284   
  285       actOutFieldInfo[i].pslf = (char *)mxGetData(actOutFieldInfo[i].pmxa);
  286   
  287       // create and attach child data out struct
  288       dims[1] = 1;
  289       paOutStruct = mxCreateStructArray(2, dims, 1, outStructFieldNames);
  290       if (paOutStruct == NULL) {
  291         m_utils.printErr(TLMG_ERR_MAT_INIT,
  292                          "## ERROR: Could not create mx data out struct for MAT file %s\n",
  293                          m_filename);
  294         exit(-1);                        //return(1);
  295       }
  296   
  297       mxSetFieldByNumber(paOutStruct, 0, 0, actOutFieldInfo[i].pmxa);
  298       mxSetFieldByNumber(m_sclog, 0, i, paOutStruct);
  299     }
  300   }

  316   {
  317     m_sclog = matGetVariable(m_pmat, m_matVar);

  319       printf("## ERROR: could not get %s variable in file %s\n",

  325       printf("## ERROR: var %s was not a struct.\n", m_matVar);
  326       return -1;

  329     return 0;

  381   // ------------------------------------------------------------------------
  382   bool mw_vecmat_tb::getNextInput()
  383   {
  384     bool endOfVec = false;
  385   
  386     // get stimulus data from MAT and put into rU.
  387     for (int i=0; i<m_matInfoIn.m_numInFields; i++) {
  388       int offset = m_vecCountIn*m_inFieldInfo[i].cfsize;
  389       if (offset+m_inFieldInfo[i].cfsize-1 > m_inFieldInfo[i].sldsize) {
  390         m_utils.printMsg(TLMG_PRINT_VEC_NUM,
  391                          "## end of input vectors in MAT file %s for input number %d\n",
  392                          m_matInfoIn.m_filename, i);
  393         endOfVec = true;
  394       } else {
  395         memcpy(m_inFieldInfo[i].pcf, m_inFieldInfo[i].pslf+offset, m_inFieldInfo[i]
  396                .
  397                cfsize);
  398       }
  399     }
  400   
  401     if (!endOfVec) {
  402       m_utils.printMsg(TLMG_PRINT_VEC_NUM,
  403                        "## vec %5d:\n##\t initialized input dataset\n",
  404                        m_vecCountIn);
  405       m_vecCountIn++;
  406       if ((!((m_vecCountIn-1) % 100)) && (m_vecCountIn != 1)) {
  407         m_utils.printMsg(TLMG_PRINT_VEC_PROGRESS_DOT, ".\n");
  408       }
  409     }
  410   
  411     return endOfVec;
  412   }

  414   // ------------------------------------------------------------------------
  415   bool mw_vecmat_tb::getNextExpectedOutput()
  416   {
  417     bool endOfVec = false;
  418     for (int i=0; i<m_matInfoIn.m_numOutFields; i++) {
  419       int offset = m_vecCountOut*m_expOutFieldInfo[i].cfsize;
  420       if (offset+m_expOutFieldInfo[i].cfsize == m_expOutFieldInfo[i].sldsize) {

  425       } else if (offset+m_expOutFieldInfo[i].cfsize > m_expOutFieldInfo[i].sldsize)
  426       {
  427         // error
  428         endOfVec = true;
  429         return endOfVec;
  430       }
  431   
  432       memcpy(m_expOutFieldInfo[i].pcf, m_expOutFieldInfo[i].pslf+offset,

  464   // ------------------------------------------------------------------------
  465   void mw_vecmat_tb::compareResult()
  466   {
  467     bool hadMiscompare = false;
  468     for (int i=0; i<m_matInfoIn.m_numOutFields; i++) {
  469       if (memcmp(m_expOutFieldInfo[i].pcf, m_actOutFieldInfo[i].pcf,
  470                  m_expOutFieldInfo[i].cfsize)) {
  471         m_utils.printErr(TLMG_ERR_MISCOMPARE,
  472                          "## ERROR: vec %d miscompare at output %d\n",
  473                          m_vecCountOut, i);
  474         hadMiscompare = true;
  475   
  476         //printf("## vector %d, field %d\n", m_vecCountOut, i);
  477         for (unsigned int j=0; j<m_expOutFieldInfo[i].cfsize; j++) {
  478           m_utils.printMsg(TLMG_PRINT_VEC_BYTE_BUF_ON_MISCOMPARE,
  479                            "\tbyte %3d: exp=%2x act=%2x\n", j,
  480                            (*(m_expOutFieldInfo[i].pcf+j)&0xff),
  481                            (*(m_actOutFieldInfo[i].pcf+j)&0xff));
  482         }
  483       }
  484     }
  485   
  486     if (hadMiscompare) {
  487       if (m_vecMiscompares++ == TLMG_MAX_NUM_MISCOMPARES) {
  488         m_utils.turnOffMsgKind(TLMG_PRINT_VEC_BYTE_BUF_ON_MISCOMPARE);
  489       }
  490   
  491       m_utils.printMsg(TLMG_PRINT_VEC_COMPARE_OK,
  492                        "##\t actual output dataset did not match expected output dataset\n");
  493     } else {
  494       m_utils.printMsg(TLMG_PRINT_VEC_COMPARE_OK,
  495                        "##\t actual output dataset matched expected output dataset\n");
  496     }
  497   
  498     m_vecCountOut++;
  499   }

  803   // ------------------------------------------------------------------------
  804   void mw_tr_tb::sendTlmWriteTransaction(sc_dt::uint64 addr, unsigned char *dptr,
  805     size_t dsize)
  806   {
  807     sc_time delay = m_sync.getLocalTime();
  808     tlm::tlm_generic_payload gp;
  809     gp.set_address(addr);
  810     gp.set_command(tlm::TLM_WRITE_COMMAND);
  811     gp.set_data_ptr(dptr);
  812     gp.set_data_length(dsize);
  813     m_socket->b_transport(gp, delay);
  814     tlm::tlm_response_status gp_status = gp.get_response_status();
  815     if (gp_status != tlm::TLM_OK_RESPONSE) {
  816       m_utils.printErr(TLMG_ERR_TRANSPORT, "## ERROR: TLM write error.\n");
  817     } else {
  818       m_utils.printMsg(TLMG_PRINT_TRANSPORT,
  819                        "##\t TLM write transaction to target component at address %#08x\n",
  820                        (unsigned int) addr);
  821       if (addr == (unsigned int) COMSTAT_REG_ADDR) {
  822         m_utils.printMsg(TLMG_PRINT_CSR_TRANSPORT,
  823                          "##\t (Command/Status reg data value = %#010x)\n",
  824                          *((unsigned int *) dptr));
  825       }
  826     }
  827   
  828     m_sync.setLocalTime(delay);
  829     m_sync.incLocalTime(SC_ZERO_TIME);
  830   }

  880   void mw_csr_tb::readModifyWrite(mw_comstatreg_tb bitMask, mw_comstatreg_tb
  881     bitValue)
  882   {
  883     m_mutex.lock();
  884   
  885     // read comstat reg
  886     m_tr.sendTlmReadTransaction(m_csrAddr, (unsigned char *) & m_csreg, sizeof
  887       (m_csreg));
  888   
  889     // modify
  890     m_csreg &= ~bitMask;                 // zero out bits of interest
  891     m_csreg |= bitValue;                 // set bits of interest
  892   
  893     // write
  894     m_tr.sendTlmWriteTransaction(m_csrAddr, (unsigned char *) & m_csreg, sizeof
  895       (m_csreg));
  896     m_mutex.unlock();
  897   }

  908   mw_comstatreg_tb mw_csr_tb::readStatus()
  909   {

  915   // ------------------------------------------------------------------------
  916   // e.g., looking for inbuf not empty or outbuf full
  917   // pollForStatus((INBUF_EMPTY_BIT_MASK | OUTBUF_FULL_BIT_MASK),
  918   //                OUTBUF_FULL_BIT_MASK, x,x,x)
  919   //
  920   // returns true on timeout, false if saw expected status.
  921   //
  922   bool mw_csr_tb::pollForStatus(mw_comstatreg_tb bitMask,
  923     mw_comstatreg_tb bitValue,
  924     mw_comstatreg_tb & comstatreg,
  925     int maxIters, sc_time iterWait)
  926   {
  927     m_sync.syncToLocalTime();            // ensure read current status
  928     m_csreg = readStatus();
  929     mw_comstatreg_tb pollResult = (m_csreg & bitMask);
  930     while ((pollResult != bitValue) && (maxIters > 0) ) {
  931       m_sync.syncToLocalTime();
  932       m_sync.syncToExplicitTime(iterWait);
  933       m_csreg = readStatus();
  934       pollResult = m_csreg & bitMask;
  935       maxIters--;
  936     }
  937   
  938     m_sync.syncToLocalTime();
  939     comstatreg = m_csreg;
  940     if (maxIters <= 0) {
  941       m_utils.printErr(TLMG_ERR_SYNC_TIMEOUT,
  942                        "## ERROR: Timed out waiting for bitValue=%#010x. (bitMask=%#010x, masked csr=%#010x)\n",
  943                        bitValue, bitMask, pollResult);
  944       return true;
  945     } else {
  946       m_utils.printMsg(TLMG_PRINT_SYNC_TO_FUNCS,
  947                        "##\t...saw polling return expected value. (bitValue=%#010x, bitMask=%#010x, masked csr=%#010x)\n",
  948                        bitValue, bitMask, pollResult);
  949       return false;
  950     }
  951   }
